# map 활용하기
   
```cpp

#include <string>
#include <vector>
#include <map>
using namespace std;

map <char, int> m = {{'w', 1}, {'s', -1}, {'d', 10}, {'a', -10}};

//미리 맵으로 문자열/이용할 숫자를 세팅 해놓기 

int solution(int n, string control)
{
    int answer = n;
    for(char ch : control)
	// string 순회
        answer += m[ch];
    return answer;
}

```
